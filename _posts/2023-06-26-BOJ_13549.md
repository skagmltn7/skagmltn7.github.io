import sys; input = sys.stdin.readline
from heapq import heappush,heappop

def bfs():
    move[n] = 0
    heap = []
    heappush(heap,(0, n))

    while heap:
        cnt, cur = heappop(heap)
        if cur == k: 
            print(cnt)
            return
        for opt, c in [(cur+1,1), (cur-1,1), (2*cur,0)]:
            if 0 <= opt <= 100000:
                clock = cnt + c
                if move[opt] > clock:
                    move[opt] = clock
                    heappush(heap,(clock,opt))

n, k = map(int,input().split())
move = [float('inf')] * 100001
bfs()
