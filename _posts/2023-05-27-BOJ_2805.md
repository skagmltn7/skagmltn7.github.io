n,m = map(int,input().split())
trees = list(map(int,input().split()))
start, end = 0, max(trees)
answer = 0
while start <= end:
    mid = (start + end)//2
    cnt = 0
    for tree in trees:
        if tree > mid: cnt += tree - mid
    
    if cnt < m: end = mid - 1
    else: 
        answer = mid
        start = mid + 1
print(answer)
