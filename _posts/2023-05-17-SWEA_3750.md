def func(num):
    tmp=0
    while num:
        tmp += num%10
        num//=10
    num=tmp
    if num//10==0: return num
    return func(num)

T = int(input())
li=[]
for tc in range(1,T+1):
    li.append(int(input()))

for i, num in enumerate(li):
    answer = func(num)
    print('#{} {}'.format(i+1,answer))
