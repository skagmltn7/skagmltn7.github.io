```python
import sys; input = sys.stdin.readline
n, m = map(int,input().split())
arr = sorted(list(map(int,input().split())))
tmp = []
chk = set()

def isDuplicate(li):
    if li not in chk: 
        chk.add(li)
        return True
    return False

def dfs(inx):
    if len(tmp) == m:
        print(' '.join(map(str,tmp)))
        return
    for i in range(inx,n):
        tmp.append(arr[i])
        if isDuplicate(tuple(tmp)):
            dfs(i)
        tmp.pop()
dfs(0)
```
