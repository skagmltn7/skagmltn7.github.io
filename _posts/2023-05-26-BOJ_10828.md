import sys
input=sys.stdin.readline

stack=[]
N = int(input())
for _ in range(N):
    command = input().split()
    if command[0]=='push': 
        stack.append(int(command[1]))
    elif command[0]=='size': 
        print(len(stack))
    else:
        if stack:
            if command[0] == 'pop': print(stack.pop())
            elif command[0] == 'empty': print(0)
            else: print(stack[-1])
        else:
            if command[0] == 'empty': print(1)
            else: print(-1)
