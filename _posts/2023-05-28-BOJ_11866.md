from collections import deque
answer=[]

n,k = map(int,input().split())
queue = deque([i for i in range(1,n+1)])

cnt = k
while queue:
    cnt-=1
    queue.append(queue.popleft())
    if cnt==0: 
        cnt = k
        answer.append(queue.pop())

print('<'+', '.join(map(str,answer))+'>')
