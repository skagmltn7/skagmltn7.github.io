import sys
input=sys.stdin.readline
answer=0

def dfs(li,inx):
    global answer
    if len(li)==3:
        li_sum = sum(li)
        if li_sum <= m and answer < li_sum : answer = li_sum
        return 
    for i in range(inx+1,n):
        if  not visited[i]:
            visited[i]=True
            li.append(numbers[i])
            dfs(li,i)
            li.pop()
            visited[i]=False

n,m = map(int,input().split())
visited = [False]*n
numbers = list(map(int,input().split()))

for i,num in enumerate(numbers):
    visited[i]=True
    dfs([num],i)
print(answer)
